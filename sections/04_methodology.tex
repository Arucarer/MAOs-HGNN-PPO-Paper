\subsection{PPO-Based Adaptive Rule Selection Framework}
\label{subsec:ppo_policy}

% 定义基于PPO的调度决策策略。
% 描述Actor网络和Critic网络的输入输出形式、

\begin{equation}
  L^{\mathrm{clip}}(\theta)=
  \mathbb{E}_t\!\left[
  \min\!\left(\rho_t(\theta)\hat{A}_t,
  \operatorname{clip}(\rho_t(\theta),1-\epsilon,1+\epsilon)\hat{A}_t\right)
  \right].
  \label{eq:ppo_clip}
\end{equation}
